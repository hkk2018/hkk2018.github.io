# Pass Params Into console.log From Wrapper Function

function failedConsoleLog() {
    console.log(arguments);
}

function justConsoleLog() {
    console.log.apply(null, arguments);
}

failedConsoleLog(1, 2, 3);
justConsoleLog(1, 2, 3);
console.log(1, 2, 3);
